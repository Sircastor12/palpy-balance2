#####rocketlauncher order
defined_text = {
	name = rocketlauncher_stock
	text = {
		trigger = {
			check_variable = { 
				var = global.rocketlauncher 
				value = global.rocketlauncherq
				compare = greater_than_or_equals
			}
		}
		localization_key = diamond_cruiser_instock
	}
	text = {
		trigger = {
			check_variable = { global.rocketlauncher < global.rocketlauncherq }
		}
		localization_key = diamond_cruiser_outstock
	}
}
defined_text = {
	name = rocketlauncher_price
	text = {
		trigger = {
			ROOT = {
				check_variable = { dx > rocketlauncher_pricem }
			}
		}
		localization_key = rocketlauncher_pricemet
	}
	text = {
		trigger = {
			ROOT = {
				check_variable = { dx < rocketlauncher_pricem }
			}
		}
		localization_key = rocketlauncher_pricenotmet
	}
}
defined_text = {
	name = rocketlauncher_tech
	text = {
		trigger = {
			ROOT = {
				has_tech = cal_weapons
			}
		}
		localization_key = rocketlauncher_techmet
	}
	text = {
		trigger = {
			ROOT = {
				NOT = {
					has_tech = cal_weapons
				}
			}
		}
		localization_key = rocketlauncher_technotmet
	}
}

	name = vulture_stock
	text = {
		trigger = {
			check_variable = { 
				var = global.vulture
				value = global.vultureq
				compare = greater_than_or_equals
			}
		}
		localization_key = vulture_instock
	}
	text = {
		trigger = {
			check_variable = { global.vulture < global.vultureq }
		}
		localization_key = vulture_outstock
	}
}
defined_text = {
	name = vulture_price
	text = {
		trigger = {
			ROOT = {
				check_variable = { dx > vulture_pricem }
			}
		}
		localization_key = vulture_pricemet
	}
	text = {
		trigger = {
			ROOT = {
				check_variable = { dx < vulture_pricem }
			}
		}
		localization_key = vulture_pricenotmet
	}
}
defined_text = {
	name = vulture_tech
	text = {
		trigger = {
			ROOT = {
				OR = {
					has_idea = CIS_member
					has_idea = CIS_member_prewar
					has_idea = CIS_equipped
					TAG = CIS
					owns_state = 22
				}
			}
		}
		localization_key = phase_1_techmet
	}
	text = {
		trigger = {
			ROOT = {
				NOT = {
					OR = {
						has_idea = CIS_member
					has_idea = CIS_member_prewar
						has_idea = CIS_equipped
						TAG = CIS
						owns_state = 22
					}
				}
			}
		}
		localization_key = phase_1_technotmet
	}
}
