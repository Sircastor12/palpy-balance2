default_unit_training = {
	enable = {
		AND = {
			NOT = {
				OR = {
					#unrestricted or custom armies
					tag = REP
					tag = CIS
					tag = IBC
					tag = GEO
					tag = FED
					tag = TEC

		}
	}
	abort = {
		always = no
	}
	ai_strategy = {
		type = build_army
		id = infantry
		value = -100
	}
}
