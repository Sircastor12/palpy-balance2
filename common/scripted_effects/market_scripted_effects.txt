purchase_rocketlauncher = {
	if = {
		limit = {
			check_variable = {
				var = global.rocketlauncher 
				value = global.rocketlauncherq
				compare = greater_than_or_equals
			}
			ROOT = {
				check_variable = { dx > rocketlauncher_pricem }
			}
			ROOT = {
				has_tech = rocketlauncher_tech
			}
		}
		Root = {
			add_equipment_to_stockpile = {
				type = rocketlauncher_equipment
				 amount = global.rocketlauncherq
			 }
			 subtract_from_variable = { dx = rocketlauncher_pricem }
			}
		subtract_from_variable = { global.rocketlauncher = global.rocketlauncherq}
	}
	if = {
		limit = {
			check_variable = { 
				var = global.rocketlauncher 
				value = global.rocketlauncherq
				compare = greater_than_or_equals
			}
			ROOT = {
				check_variable = { dx > rocketlauncher_pricem }
			}
			ROOT = {
				NOT = {
					has_tech = rocketlauncher_tech
				}
			}
		}
		Root = {
			set_technology = {
				rocketlauncher_tech = 1
				rocketlauncher_unit_tech = 1
			}
			add_equipment_to_stockpile = {
				type = rocketlauncher_equipment

			 }

}
